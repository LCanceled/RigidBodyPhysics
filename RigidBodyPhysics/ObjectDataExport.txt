
import bpy

out = open('C:\\Users\\Aric\\Documents\\Bluetooth Exchange Folder\\info.txt', 'w')

for ob in bpy.data.objects:
    if (ob.type == 'MESH'):
        quat = ob.rotation_euler.to_quaternion()
        out.write('%s %f %f %f %f %f %f %f\n' % (ob.name, ob.location.x, ob.location.y, ob.location.z, quat.x, quat.y, quat.z, quat.w))
out.close()
